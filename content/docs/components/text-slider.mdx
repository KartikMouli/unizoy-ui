```tsx
import clsx, { ClassValue } from 'clsx'
import { twMerge } from 'tailwind-merge'
export const cn = (...classes: ClassValue[]) => twMerge(clsx(...classes))

export  function mergeRefs<T>(...refs: (React.Ref<T> | undefined)[]) {
  return (node: T | null) => {
    refs.forEach((ref) => {
      if (typeof ref === "function") {
        ref(node);
      } else if (ref && "current" in ref) {
        (ref as React.MutableRefObject<T | null>).current = node;
      }
    });
  };
}

```

### Props

| Name           | Type                                                  | Description                                                            |
| -------------- | ----------------------------------------------------- | ---------------------------------------------------------------------- |
| `direction`    | `"up"\|"down"`                                        | Direction of the slide.                                                |
| `duration`     | `number`                                              | Duration of the animation.                                             |
| `ease`         | `string`                                              | Animation style, [refers to gsap docs](https://gsap.com/docs/v3/Eases) |
| `stagger`      | `number`                                              | Delay between two consecuetive element's animation `seconds`.       |
| `start`        | `string` or `number` or `()=>string\|number`          | Animation start point, [refers to gsap docs](https://gsap.com/docs/v3/Plugins/ScrollTrigger/?page=1#start)  |
| `end`          | `string` or `number` or `()=>string\|number`          | Animation end point, [refers to gsap docs](https://gsap.com/docs/v3/Plugins/ScrollTrigger/?page=1#end)  |
